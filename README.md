## Install

```bash
npm install --save react-finch-connect
```

## Usage

```jsx
import React, { Component } from 'react'

import { useMyHook } from 'react-finch-connect'

const Example = () => {
  const example = useMyHook()
  return (
    <div>{example}</div>
  )
}
```
